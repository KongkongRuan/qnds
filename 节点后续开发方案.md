
#### 3.2.3 核心实现

```java
@Component
public class SnmpTrapReceiver implements CommandResponder {

    private static final Logger log = LoggerFactory.getLogger(SnmpTrapReceiver.class);
    
    private Snmp snmp;
    private int listenPort = 162;  // 默认端口，可配置
    
    @Autowired
    private AlarmForwardService alarmForwardService;
    
    @Autowired
    private DeviceCollectorService deviceCollectorService;
    
    @PostConstruct
    public void init() throws IOException {
        // 监听UDP端口
        Address listenAddress = GenericAddress.parse("udp:0.0.0.0/" + listenPort);
        TransportMapping<?> transport = new DefaultUdpTransportMapping(
            (UdpAddress) listenAddress);
        
        snmp = new Snmp(transport);
        
        // 添加CommandResponder处理Trap
        snmp.addCommandResponder(this);
        
        // 启动监听
        transport.listen();
        log.info("SNMP Trap接收器已启动，监听端口: {}", listenPort);
    }
    
    @Override
    public void processPdu(CommandResponderEvent event) {
        PDU pdu = event.getPDU();
        if (pdu == null) {
            return;
        }
        
        // 获取发送方地址
        Address peerAddress = event.getPeerAddress();
        String deviceIp = ((UdpAddress) peerAddress).getInetAddress().getHostAddress();
        
        log.info("收到Trap，来源: {}, 类型: {}", deviceIp, pdu.getType());
        
        // 解析Trap
        AlarmInfo alarm = parseTrap(deviceIp, pdu);
        
        // 转发告警
        if (alarm != null) {
            alarmForwardService.forwardAlarm(alarm);
        }
    }
    
    private AlarmInfo parseTrap(String deviceIp, PDU pdu) {
        AlarmInfo alarm = new AlarmInfo();
        
        // 根据设备IP查找设备ID
        String deviceId = deviceCollectorService.getDeviceIdByIp(deviceIp);
        alarm.setAlarmSource(deviceId);
        
        // 解析Trap变量绑定
        for (VariableBinding vb : pdu.getVariableBindings()) {
            String oid = vb.getOid().toString();
            String value = vb.getVariable().toString();
            
            // 根据OID映射告警码
            String alarmCode = mapOidToAlarmCode(oid);
            if (alarmCode != null) {
                alarm.setAlarmCode(alarmCode);
                alarm.setAlarmData(value);
            }
        }
        
        // 设置告警时间
        alarm.setAlarmCreateDate(LocalDateTime.now()
            .format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss")));
        
        return alarm;
    }
    
    @PreDestroy
    public void destroy() throws IOException {
        if (snmp != null) {
            snmp.close();
        }
        log.info("SNMP Trap接收器已关闭");
    }
}
```

```java
@Service
public class AlarmForwardService {
    
    private static final Logger log = LoggerFactory.getLogger(AlarmForwardService.class);
    
    @Autowired
    private RestTemplate restTemplate;
    
    @Value("${alarm.platform-url}")
    private String platformUrl;
    
    @Value("${alarm.enabled:true}")
    private boolean enabled;
    
    @Value("${alarm.retry-count:3}")
    private int retryCount;
    
    /**
     * 转发告警到管控平台
     */
    public void forwardAlarm(AlarmInfo alarm) {
        if (!enabled) {
            log.debug("告警转发已禁用");
            return;
        }
        
        for (int i = 0; i < retryCount; i++) {
            try {
                ResponseEntity<Map> response = restTemplate.postForEntity(
                    platformUrl, alarm, Map.class);
                
                if (response.getStatusCode().is2xxSuccessful()) {
                    log.info("告警转发成功: 设备={}, 告警码={}", 
                        alarm.getAlarmSource(), alarm.getAlarmCode());
                    return;
                }
            } catch (Exception e) {
                log.warn("告警转发失败(第{}次): {}", i + 1, e.getMessage());
                
                if (i < retryCount - 1) {
                    try {
                        Thread.sleep(1000 * (i + 1));  // 递增等待
                    } catch (InterruptedException ie) {
                        Thread.currentThread().interrupt();
                    }
                }
            }
        }
        
        // 重试失败，可考虑本地缓存
        log.error("告警转发最终失败: {}", alarm);
        cacheFailedAlarm(alarm);
    }
    
    /**
     * 缓存转发失败的告警（可选实现）
     */
    private void cacheFailedAlarm(AlarmInfo alarm) {
        // TODO: 实现本地缓存，后续重试发送
    }
}
```

```java
@RestController
@RequestMapping("/api/config")
public class ConfigController {
    
    private static final Logger log = LoggerFactory.getLogger(ConfigController.class);
    
    @Autowired
    private SnmpClient snmpClient;
    
    @Autowired
    private DeviceCollectorService deviceCollectorService;
    
    /**
     * 配置下发接口
     */
    @PostMapping("/set")
    public ResponseEntity<ConfigResponse> setConfig(@RequestBody ConfigRequest request) {
        ConfigResponse response = new ConfigResponse();
        response.setDeviceId(request.getDeviceId());
        
        // 获取设备信息
        DeviceInfo device = deviceCollectorService.getDeviceById(request.getDeviceId());
        if (device == null) {
            response.setCode(404);
            response.setMessage("设备不存在");
            return ResponseEntity.status(404).body(response);
        }
        
        List<ConfigResponse.ConfigResult> results = new ArrayList<>();
        int successCount = 0;
        
        for (ConfigRequest.ConfigItem item : request.getConfigs()) {
            ConfigResponse.ConfigResult result = new ConfigResponse.ConfigResult();
            result.setOid(item.getOid());
            
            try {
                SnmpResult snmpResult = snmpClient.set(
                    device.getDeviceIp(),
                    Integer.parseInt(device.getDevicePort()),
                    item.getOid(),
                    item.getValue(),
                    item.getType(),
                    device.getProtocol(),
                    device
                );
                
                result.setSuccess(snmpResult.isSuccess());
                result.setMessage(snmpResult.getError());
                result.setNewValue(item.getValue());
                
                if (snmpResult.isSuccess()) {
                    successCount++;
                }
            } catch (Exception e) {
                result.setSuccess(false);
                result.setMessage(e.getMessage());
            }
            
            results.add(result);
        }
        
        response.setCode(200);
        response.setMessage("配置下发完成");
        response.setTotalCount(request.getConfigs().size());
        response.setSuccessCount(successCount);
        response.setFailCount(request.getConfigs().size() - successCount);
        response.setResults(results);
        
        return ResponseEntity.ok(response);
    }
    
    /**
     * 批量配置下发（多设备）
```

```java
/**
 * 设置设备的Trap目标地址为当前节点IP

private boolean setTrapTarget(DeviceInfo device) {
    try {
        String nodeIp = InetAddress.getLocalHost().getHostAddress();
        
        // 设置Trap目标地址（OID需根据设备厂商MIB确定）
        // 示例：使用SNMP-TARGET-MIB中的snmpTargetAddrTable
        String trapTargetOid = "1.3.6.1.6.3.12.1.2.1.4.1"; // 示例OID
        
        SnmpResult result = snmpClient.set(
            device.getDeviceIp(),
            Integer.parseInt(device.getDevicePort()),
            trapTargetOid,
            nodeIp,
            "IPADDRESS",
            device.getProtocol(),
            device
        );
        
        if (result.isSuccess()) {
            log.info("设备Trap目标设置成功: deviceId={}, targetIp={}", 
                device.getId(), nodeIp);
            return true;
        } else {
            log.warn("设备Trap目标设置失败: deviceId={}, error={}", 
                device.getId(), result.getError());
            return false;
        }
    } catch (Exception e) {
        log.error("设置Trap目标异常: deviceId={}", device.getId(), e);
        return false;
    }
}
```

```java
@Service
public class AlarmCodeMappingService {
    
    @Value("#{${alarm.code-mapping}}")
    private Map<String, String> codeMapping;
    
    /**
     * 根据Trap OID获取告警码
     */
    public String getAlarmCode(String trapOid) {
        if (trapOid == null) {
            return "UNKNOWN";
        }
        
        // 精确匹配
        if (codeMapping.containsKey(trapOid)) {
            return codeMapping.get(trapOid);
        }
        
        // 前缀匹配
        for (Map.Entry<String, String> entry : codeMapping.entrySet()) {
            if (trapOid.startsWith(entry.getKey())) {
                return entry.getValue();
            }
        }
        
        return "UNKNOWN";
    }
}
```
